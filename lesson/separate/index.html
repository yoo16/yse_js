HTML

セレクタ


      $(セレクタ).メソッド(パラメータ);
      //または
      jQuery(セレクタ).メソッド(パラメータ);

val() メソッド


      //値を設定
      $(セレクタ).val(パラメータ);

      //値を取得
      $(セレクタ).val();

DOMContentLoaded

  • DOM読み込み完了後に、処理を実行する
  • DOM読み込み前にJavaScriptが実行されると、動作しない場合が多い


      $(function () {
        //DOM読み込み完了時に、実行される
      }

      //古い書き方
      $(document).ready(function() {
        //DOM読み込み完了時に、実行される
      });

変数

JS:変数を利用しない場合


      $('#itemList li').removeClass('dummy'); //class から dummy を削除
      $('#itemList li').addClass('item');     //class に item を追加
      $('#itemList li').css('color', 'red');  //css に color="red" を追加

JS:変数を利用する場合


      let target = $('#itemList li');
      target.removeClass('item');
      target.addClass('item');
      target.css('color', 'red');

JS:省略（メソッドチェーン）


      let target = $('#itemList li');
      target.removeClass('item').addClass('item').css('color', 'red');
